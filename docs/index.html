p.turtle

Basic LOGO turtle implemented in p5.js.

Minimal example


  var sketch = function(p) {
    var turtle, run;

    p.setup = function() {
      p.createCanvas(400, 400);
      
      p.background(200);
      p.angleMode(p.DEGREES);
      p.strokeWeight(1);

      turtle = new Turtle(p);
      turtle.penDown = true;

      for (var i = 0; i < 6; i++) {
        turtle.color = 'black';
        turtle.forward(50);
        turtle.right(360 / 6);
      }

      turtle.right(360);
      
      run = turtle.getRun();
    };

    p.draw = function() {
      p.background(200);
      run.animate(2);
    };
  };

  var minimal = new p5(sketch);
